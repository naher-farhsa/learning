//  List -> Map -> List
import java.util.stream.*;
import java.util.*;
class Main {
    public static void main(String[] args) {
        System.out.println("Try programiz.pro");
        
    List<Integer> l=Arrays.asList(1,2,0,2,5,4,5,1,0,0);
        
     Map<Integer,Integer> m=l.stream()
                           .collect(Collectors
                           .toMap(k->k, v->1, (u,v)-> u+v)
                           );
    System.out.println(m);                       
     List<Integer> dl=m.keySet()
                       .stream()
                       .filter(k->m.get(k)>1)
                       .collect(Collectors.toList());
    System.out.println(dl);                  
        
    }
}


Note: 
reduce - BiFunction  - Works on binary operators apply (compare, add, sub, mul, div)
map - Function -  Works on Unary operators apply (increament/decreament, condition chck (even,odd,etc))
Collectors.toMap(keyMapper, valueMapper, mergeFunction)
